Setup

##Clear the environment
rm(list=ls())

# Turn off scientific notations for numbers
options(scipen = 999)  

#Set locale
Sys.setlocale("LC_ALL", "English") 
## [1] "LC_COLLATE=English_United States.1252;LC_CTYPE=English_United States.1252;LC_MONETARY=English_United States.1252;LC_NUMERIC=C;LC_TIME=English_United States.1252"
#Set variables to use
set.seed(2345) ##Set the seed for reproducibility
nbr = 2000 ##Number of observations for each group (3 groups)
sampsize = nbr #Size of samples
nbrsamples = 10000 #how many samples
withreplace = 1 #With replacement 0 or 1 (1 for bootstrap) 
conf = .95 ##COnfidence Interval

Generate random data

##Create the data attributes
##Speed
s1<- rnorm(nbr,70,15)
s2<- rep(NA,nbr)
s3<- rnorm(nbr,20,4)
speed<-c(s1,s2,s3)
rm(s1,s2,s3)

##Cycles
c1<- rbinom(nbr,3,.6)
c2<- rbinom(nbr,3,.4)
c3<- rep(NA,nbr)
cycle<-c(c1,c2,c3)
rm(c1,c2,c3)

##Wear
w1<- rep(NA,nbr)
w2<- rchisq(nbr,4)
w3<- rchisq(nbr,7)
wear<-c(w1,w2,w3)
rm(w1,w2,w3)

##Establish Group Numbers
group<- c(rep(1,nbr),rep(2,nbr),rep(3,nbr))

##Make a dataframe
df <-data.frame(speed=speed, cycle=cycle, wear=wear, group=group)
rm(speed,cycle,wear,group)

resamples <-matrix(sample(df$speed,sampsize*nbrsamples,replace=withreplace),nbrsamples,sampsize)
means_speed <-apply (resamples,1,mean, na.rm=1)
hist(means_speed, main=maintitle)
rug(means_speed)
abline(v=mean(df$speed,na.rm=1),col="red")
